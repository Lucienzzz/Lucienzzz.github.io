1.算法复杂度

​ 时间复杂度用大o表示，没什么好说的。主要来说下递归的算法复杂度

    T(N) = a*T(N/b) + O(N^d)
    1) log(b,a) > d -> 复杂度为O(N^log(b,a))
    2) log(b,a) = d -> 复杂度为O(N^d * logN)
    3) log(b,a) < d -> 复杂度为O(N^d)

​ 补充阅读： www.gocalf.com/blog/algorithm-complexity-and-mastertheorem.html （虽然我就没看过，哈哈哈哈）

​ 举个例子来说一下

    T(N) = 2*T(N/2) + O(N^1)
    log2 2 = 1 ->复杂度是o（N*logN）

2.归并排序

​ 归并排序的思想是分而治之，把一个大问题转换为同类型的小问题，同类型指的是运算过程，运算逻辑相等，比如说，我想给N个数排序，那么我把这个过程分为三部分，第一部分，我先给0~mid部分排序，然后给mid+1~N排序，然后把两个排好序的序列，再合起来排序，这样就达到了给一个序列排序的过程，类似的，2分之后还有4分，知道一个数就返回。这就是归并排序的思路。

public class 归并排序 {
    public static void main(String[] args) {
        int a[] = {1,-2,-3,56,3};
        归并排序算法(a,0,a.length-1);
        for (int i = 0; i < a.length; i++) {
            System.out.println(a[i] + " ");
        }
    }
    public static void 归并排序算法(int a[], int L, int R){
        if(L == R)
            return;
        int mid = L + ((R - L) >> 1);//防止溢出
        归并排序算法(a,L,mid);
        //错误点1：这边不是0,记住是L到mid，因为这个每一次都不一定是L开始的
        归并排序算法(a,mid+1,R);
        整合(a, L, R, mid);
    }

    private static void 整合(int a[], int L, int R, int mid) {
        int help[] = new int[R - L + 1];
        //错误点2，这边不是a.length,这边应该是L-R+1个个数长度，为什么呢？实际debug表明：如果是a长度的话，
        // java会自动生成a长度大小，元素全为0的数组，在最后的覆盖中会把后面那些还没有排好序的元素也替换成0，污染了源数据
        int count = 0;
        int p1 = L;//不是0是L
        int p2 = mid + 1;
        while (p1 <= mid && p2 <= R){
            help[count++] = a[p1] <= a[p2]? a[p1++] : a[p2++];
        }
        while (p1 <= mid){
            help[count++] = a[p1++];
        }
        while (p2 <= R){
            help[count++] = a[p2++];
        }
        for (int i = 0; i < help.length; i++) {
            a[L + i] = help[i];
            //错误点三：这边不是a[i] = help[i];，因为这样做得话，每一次就总会从替换第0个数据，一直覆盖源数据，导致错误
            //正确的办法应该是从L开始，因为L是每一次源数据的开始，L+i就是每一次源数据所在位置的加减。
        }
    }
}

3.冒泡排序（自己实现的）

    public static void 冒泡排序(int a[]){
        //如果左边开始数 第一个大于第二个  就开始交换， 知道最后一个 我们就能得到升序排列，否则就是降序排列
        for (int i = 0; i < a.length-1; i++) {
            for (int j = 0; j < a.length-1-i; j++) {
                if(a[j]>a[j+1]){
                    int temp = a[j];
                    a[j] = a[j+1];
                    a[j+1] = temp;
                }
            }
        }
    }

4.选择排序（自己实现的）

    public static void 选择排序(int a[]){
        ;
        for (int i = 0; i < a.length - 1; i++) {
            int Min = a[i];
            for (int j = i; j < a.length; j++) {
                if(Min > a[j]){
                    int temp = Min;
                    Min = a[j];
                    a[j] = temp;
                }
            }
            a[i] = Min;
        }
    }

5.插入排序（自己实现的）

    public static void 插入排序(int a[]){
        //一个数据分为两个部分，排好序的有序序列，和无序序列，每次向有序序列中添加一个数字即可。
        for (int i = 0; i < a.length - 1; i++) {
            for (int j = i; j >= 0; j--) {
                if(a[j+1] < a[j]){
                    int temp = a[j+1];
                    a[j+1] = a[j];
                    a[j] = temp;
                }
            }
        }
    }

6.小合问题

public class 小合问题 {
    /**
     * 在一个数组中， 每一个数左边比当前数小的数累加起来， 叫做这个数组的小和。 求一个数组的小和。
     * 例子：
     * [1,3,4,2,5]
     * 1左边比1小的数， 没有；
     * 3左边比3小的数， 1；
     * 4左边比4小的数， 1、 3；
     * 2左边比2小的数， 1；
     * 5左边比5小的数， 1、 3、 4、 2；
     * 所以小和为1+1+3+1+1+3+4+2=16
     */
    /**
     * 思路：可以转化为寻找比左边数字大的元素个数*该左边元素即可
     */
    public static void main(String[] args) {
        litterSum();
    }

    private static void litterSum() {
        int a[] = {1,3,4,2,5};
        System.out.println(solveLitterSum(a, 0, a.length - 1));
//        for (int i = 0; i < a.length; i++) {
//            System.out.println(a[i] + " ");
//        }
    }

    private static int solveLitterSum(int[] a, int L, int R) {
        //返回值类型发生变化，以前是什么都不返回，现在返回一个整形
        if(L == R)
            return 0;
        //当两个值一样的时候，就表示碰到了同一个数，所以我们要返回0嘛
        int mid = L + ((R - L) >> 1);
//        solveLitterSum(a,L,mid);
//        solveLitterSum(a,mid+1,R);
//        merage(a,L,mid,R);
        return solveLitterSum(a,L,mid) + solveLitterSum(a,mid+1,R) + merage(a,L,mid,R);
    }

    private static int merage(int[] a, int L, int mid, int R) {
        int help[] = new int[R - L + 1];
        int count = 0;
        int p1 = L;
        int p2 = mid + 1;
        int res = 0;
        while (p1 <= mid && p2 <= R){
            res += a[p1] < a[p2] ? (R - p2 + 1)*a[p1]:0;
            help[count++] = a[p1] <= a[p2]? a[p1++] : a[p2++];
        }
        while (p1 <= mid){
            help[count++] = a[p1++];
        }
        while (p2 <= R){
            help[count++] = a[p2++];
        }
        for (int i = 0; i < help.length; i++) {
            a[L + i] = help[i];
        }
        return res;
    }
}
